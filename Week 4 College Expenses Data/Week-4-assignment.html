# I am looking to see if there is any correation between the tuition prices for each school and how they spend that money back. For this, student sercives is an important variable to look at. MOre expensive schools should theoretically have more income in student services.
col <- df %>%
  mutate(TuitionQuar = Tuition / Total.Enrollment) %>%
  mutate(ServicesQuar = Student.Services / Total.Enrollment)


ggplot(col, aes(x = TuitionQuar, y = `ServicesQuar`)) + 
  geom_point() +
  geom_smooth(method = "lm", se = FALSE) +
  scale_y_continuous(labels = label_number(scale = 1/100000, suffix = 'K')) +
  ylim(0, 20000) + 
  xlim(0, 175000)
